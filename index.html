import { state, effect, context, Router, Out, layout }
    from 'https://cdn.jsdelivr.net/npm/@agberohq/oja@latest/build/oja.core.esm.js';

// Shared reactive state — readable from any component, anywhere
const [tasks, setTasks] = state([]);
const [user,  setUser]  = context('user', null);

// Persistent shell — nav stays mounted across all routes
await layout.apply('#app', 'layouts/shell.html', { user });

const router = new Router({ outlet: '#main' });
router.Get('/',      Out.c('pages/home.html',  { tasks, user }));
router.Get('/tasks', Out.c('pages/tasks.html', { tasks }));
router.start('/');  // No build step. No compiler. Just this.

Why Oja exists

the insight that shaped it
UI developer

Opens .html and .css only

Components are plain HTML files. A UI developer can open them directly in a browser, edit them, and see results — no JSX, no compilation, no tooling required.

JS developer

Opens .js only, never writes HTML strings

The JS supplies data and wires events. It never builds HTML. The component file is the template — no template literals, no render functions, no virtual DOM.
